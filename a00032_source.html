526  void optimize_dims(int i,
527  const std::array<int, dim>& size,
528  int P,
529  std::array<int, dim>& dims,
530  std::array<int, dim>& trydims,
531  double& opt) const
532  {
533  if (i > 0) // test all subdivisions recursively
534  {
535  for (unsigned int k = 1; k <= (unsigned int)P; k++)
536  // dune-randomfield needs exact divisibility
537  if (P % k == 0 && size[i] % k == 0) {
538  // P divisible by k
539  trydims[i] = k;
540  optimize_dims(i - 1, size, P / k, dims, trydims, opt);
541  }
542  } else {
543  // dune-randomfield needs exact divisibility
544  if (size[0] % P == 0) {
545  // found a possible combination
546  trydims[0] = P;
547 
548  // check for optimality
549  double m = -1.;
550 
551  for (unsigned int k = 0; k < dim; k++) {
552  double mm = ((double)size[k]) / ((double)trydims[k]);
553  if (fmod((double)size[k], (double)trydims[k]) > 0.0001)
554  mm *= 3;
555  if (mm > m)
556  m = mm;
557  }
558  if (m < opt) {
