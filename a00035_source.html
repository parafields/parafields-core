55template<typename RF, unsigned int dim>
56void
57readParallelFromHDF5(std::vector<RF>& local_data,
58 const std::array<unsigned int, dim>& local_count,
59 const std::array<unsigned int, dim>& local_offset,
60 const MPI_Comm& communicator,
61 const std::string& data_name,
62 const std::string& data_filename)
63{
64 // setup file access template with parallel IO access
65 hid_t access_pList = H5Pcreate(H5P_FILE_ACCESS);
66 assert(access_pList > -1);
67
68 herr_t status;
69 MPI_Info mpiInfo = MPI_INFO_NULL;
70 status = H5Pset_fapl_mpio(access_pList, communicator, mpiInfo);
71 assert(status > -1);
72
73 // open the file for reading
74 hid_t file_id = H5Fopen(data_filename.c_str(), H5F_ACC_RDONLY, access_pList);
75 assert(file_id > -1);
76
77 // Release file-access template
78 status = H5Pclose(access_pList);
79 assert(status > -1);
80
81 // open the dataset
82 hid_t dataset_id = H5Dopen(file_id, data_name.c_str(), H5P_DEFAULT);
83 assert(dataset_id > -1);
84
85 // get the dataspace
86 hid_t dataspace_id = H5Dget_space(dataset_id);
87 assert(dataspace_id > -1);
88
89 // some needed variables
90 hsize_t dimData;
91 hsize_t* dims;
92
93 // get the dimension
94 dimData = H5Sget_simple_extent_ndims(dataspace_id);
95 assert(dimData == dim);
96
97 // get the size of the data structure
98 dims = (hsize_t*)malloc(dim * sizeof(hsize_t));
99 status = H5Sget_simple_extent_dims(dataspace_id, dims, 0);
100 assert(status > -1);
101
102 // set the local, offset, and count as hsize_t, which is needed by the HDF5
103 // routines
104 hsize_t local_size = 1;
105 hsize_t offset[dim], count[dim];
106 for (unsigned int i = 0; i < dim; i++) {
107 local_size *= local_count[i];
108 offset[dim - i - 1] = local_offset[i];
109 count[dim - i - 1] = local_count[i];
110 }
111
112 // create the memory space, if something needs to be read on this processor
113 hid_t memspace_id = 0;
114 if (local_size != 0)
115 memspace_id = H5Screate_simple(1, &local_size, NULL);
116
117 // select the hyperslab
118 status = H5Sselect_hyperslab(
119 dataspace_id, H5S_SELECT_SET, offset, NULL, count, NULL);
120 assert(status > -1);
121
122 // resize the return data
123 local_data.resize(local_size);
124
125 // set up the collective transfer properties list
126 hid_t xferPropList = H5Pcreate(H5P_DATASET_XFER);
127 assert(xferPropList > -1);
128
129 // finally the reading from the file, only if something needes to be read
130 if (local_size != 0) {
131 status = H5Dread(dataset_id,
132 hdf5Type<RF>,
133 memspace_id,
134 dataspace_id,
135 xferPropList,
136 &(local_data[0]));
137 assert(status > -1);
138 }
139
140 // close the identifiers
141 H5Dclose(dataset_id);
142 H5Sclose(dataspace_id);
143 if (local_size !=
144 0) // this identifier only exists if somethings needs to be read
145 H5Sclose(memspace_id);
146 free(dims);
147 status = H5Fclose(file_id);
148 assert(status > -1);
149}

169template<typename RF, unsigned int dim>
170void
171writeParallelToHDF5(const std::array<unsigned int, dim>& global_dim,
172 const std::vector<RF>& local_data,
173 const std::array<unsigned int, dim>& local_count,
174 const std::array<unsigned int, dim>& local_offset,
175 const MPI_Comm& communicator,
176 const std::string& data_name,
177 const std::string& data_filename)
178{
179 // Info variable needed for HDF5
180 MPI_Info mpiInfo = MPI_INFO_NULL;
181 herr_t status;
182
183 // Set up file access property list with parallel I/O access
184 hid_t plist_id = H5Pcreate(H5P_FILE_ACCESS);
185 H5Pset_fapl_mpio(plist_id, communicator, mpiInfo);
186 assert(plist_id > -1);
187
188 // Create a new file using default properties.
189 hid_t file_id =
190 H5Fcreate(data_filename.c_str(), H5F_ACC_TRUNC, H5P_DEFAULT, plist_id);
191 assert(file_id > -1);
192 H5Pclose(plist_id);
193
194 // set the global size of the grid into a vector of type hsize_t (needed for
195 // HDF5 routines)
196 hsize_t global_dim_HDF5[dim];
197 for (unsigned int i = 0; i < dim; i++)
198 global_dim_HDF5[dim - i - 1] = global_dim[i];
199
200 // set the count and offset in the different dimensions (determine the size of
201 // the hyperslab) (in hsize_t format, needed for HDF5 routines)
202 hsize_t count[dim], offset[dim];
203 for (unsigned int i = 0; i < dim; i++) {
204 count[dim - i - 1] = local_count[i];
205 offset[dim - i - 1] = local_offset[i];
206 }
207
208 // define the total size of the local data
209 hsize_t nAllLocalCells = 1;
210 for (unsigned int i = 0; i < dim; i++)
211 nAllLocalCells *= count[i];
212
213 // Create the dataspace for the dataset.
214 hid_t filespace = H5Screate_simple(dim, global_dim_HDF5, NULL);
215 assert(filespace > -1);
216
217 // Create the dataset with default properties and close filespace.
218 hid_t dset_id = H5Dcreate(file_id,
219 data_name.c_str(),
220 HDF5_DATA_TYPE,
221 filespace,
222 H5P_DEFAULT,
223 H5P_DEFAULT,
224 H5P_DEFAULT);
225 H5Sclose(filespace);
226 assert(dset_id > -1);
227
228 // get the memoryspace (but only if something needs to be written on this
229 // processor!)
230 hid_t memspace_id;
231 if (nAllLocalCells !=
232 0) { // -> otherwise HDF5 warning, because of writing nothing!
233 memspace_id = H5Screate_simple(dim, count, NULL);
234 assert(memspace_id > -1);
235 }
236
237 // Select hyperslab in the file
238 filespace = H5Dget_space(dset_id);
239 H5Sselect_hyperslab(filespace, H5S_SELECT_SET, offset, NULL, count, NULL);
240
241 // Create property list for collective dataset write.
242 plist_id = H5Pcreate(H5P_DATASET_XFER);
243 H5Pset_dxpl_mpio(plist_id, H5FD_MPIO_COLLECTIVE);
244
245 // finally write the data to the disk
246 // even if nothing should be written H5Dwrite needs to be called!
247 if (nAllLocalCells !=
248 0) { // -> otherwise HDF5 warning, because of writing nothing!
249 status = H5Dwrite(dset_id,
250 hdf5Type<RF>,
251 memspace_id,
252 filespace,
253 plist_id,
254 &(local_data[0]));
255 assert(status > -1);
256 } else { // IMPORTANT. otherwise the H5Dwrite() blocks!
257 status =
258 H5Dwrite(dset_id, hdf5Type<RF>, 0, filespace, plist_id, &(local_data[0]));
259 assert(status > -1);
260 }
261
262 // Close the property list
263 status = H5Pclose(plist_id);
264 assert(status > -1);
265
266 // Close the filespace
267 status = H5Sclose(filespace);
268 assert(status > -1);
269
270 // if something written close the memspace
271 if (nAllLocalCells != 0) {
272 status = H5Sclose(memspace_id);
273 assert(status > -1);
274 }
275
276 // Close the dataset
277 status = H5Dclose(dset_id);
278 assert(status > -1);
279
280 // Close the file
281 status = H5Fclose(file_id);
282 assert(status > -1);
283
284 // propably not needed. because the H5Dwrite blocks anyway
285 MPI_Barrier(communicator);
286}
