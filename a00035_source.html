585 void evaluate(const typename Traits::DomainType& location,
586 typename Traits::RangeType& output) const
587 {
588 if (TrendComponentType::isMean(componentType)) {
589 output[0] = meanVector[0] + shiftVector[0];
590 } else if (TrendComponentType::isSlope(componentType)) {
591 output[0] = 0.;
592
593 for (unsigned int i = 0; i < dim; i++)
594 output[0] +=
595 (meanVector[i] + shiftVector[i]) * (location[i] - extensions[i] / 2.);
596 } else if (TrendComponentType::isDisk(componentType)) {
597 output[0] = 0.;
598
599 RF distSquared = 0.;
600 for (unsigned int i = 0; i < dim; i++)
601 distSquared +=
602 std::pow(location[i] - (meanVector[i] + shiftVector[i]), 2);
603
604 output[0] = std::exp(-distSquared /
605 std::pow(meanVector[dim] + shiftVector[dim], 2) *
606 (meanVector[dim + 1] + shiftVector[dim + 1]));
607 } else if (TrendComponentType::isBlock(componentType)) {
608 output[0] = 0.;
609
610 for (unsigned int i = 0; i < dim; i++) {
611 if (std::abs(location[i] - (meanVector[i] + shiftVector[i])) >
612 0.5 * (meanVector[dim + i] + shiftVector[dim + i]))
613 return;
614 }
615
616 output[0] = meanVector[2 * dim] + shiftVector[2 * dim];
617 } else
618 throw std::runtime_error{ "Trend component type not found!" };
619 }
