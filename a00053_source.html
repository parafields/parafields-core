105 matrixValues[5] = copy[2] * copy[3] - copy[0] * copy[5];
106 matrixValues[6] = copy[3] * copy[7] - copy[4] * copy[6];
107 matrixValues[7] = copy[1] * copy[6] - copy[0] * copy[7];
108 matrixValues[8] = copy[0] * copy[4] - copy[1] * copy[3];
109

111 copy[1] * (copy[3] * copy[8] - copy[5] * copy[6]) +
112 copy[2] * (copy[3] * copy[7] - copy[4] * copy[6]);
113 for (unsigned int i = 0; i < dim * dim; i++)
114 matrixValues[i] /= det;
115 } else if (dim == 2) {
116 matrixValues[0] = copy[3];
117 matrixValues[1] = -copy[1];
118 matrixValues[2] = -copy[2];
119 matrixValues[3] = copy[0];
120
121 const RF det = copy[0] * copy[3] - copy[1] * copy[2];
122 for (unsigned int i = 0; i < dim * dim; i++)

147class SmoothSigmoid
148{
149public:
160 template<typename RF>
161 RF operator()(RF oneEdge,
162 RF zeroEdge,
163 RF x,
164 unsigned int recursions = 0) const
165 {
166 if (std::abs(x) < oneEdge)
167 return 1.;
168 if (std::abs(x) > zeroEdge)
169 return 0.;
170
171 const RF x1 = (zeroEdge - std::abs(x)) / (zeroEdge - oneEdge);
172 const RF x2 = (std::abs(x) - oneEdge) / (zeroEdge - oneEdge);
173
174 const RF output = theta(x1) / (theta(x1) + theta(x2));
175
176 return output;
177 }
178
179private:
187 template<typename RF>
188 RF theta(RF x) const
189 {
190 if (x > 0.)
191 return std::exp(-1. / x);
192 else
193 return 0.;
194 }
195};

296class SphericalCovariance
297{
298public:
310 template<typename RF, long unsigned int dim>
311 RF operator()(const RF variance, const std::array<RF, dim>& x) const
312 {
313 RF sum = 0.;
314 for (unsigned int i = 0; i < dim; i++)
315 sum += x[i] * x[i];
316 RF h_eff = std::sqrt(sum);
317
318 if (dim == 3) {
319 if (h_eff > 1.)
320 return 0.;
321 else
322 return variance * (1. - 1.5 * h_eff + 0.5 * std::pow(h_eff, 3));
323 } else if (dim == 2) {
324 if (h_eff > 1.)
325 return 0.;
326 else
327 return variance *
328 (1. - 2. / Dune::StandardMathematicalConstants<RF>::pi() *
329 (h_eff * std::sqrt(1. - std::pow(h_eff, 2)) +
330 std::asin(h_eff)));
331 } else if (dim == 1) {
332 if (h_eff > 1.)
333 return 0.;
334 else
335 return variance * (1. - h_eff);
336 } else
337 throw std::runtime_error{
338 "spherical covariance only defined for 1D, 2D and 3D"
339 };
340 }
