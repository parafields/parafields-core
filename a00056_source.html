366 unsigned int makePositive(Real shift,
367 Real threshold,
368 Real multiplier = 1.,
369 bool logSumExp = false)
370 {
371 unsigned int negative = 0;
372
373 if (logSumExp) {
374 for (Index i = 0; i < data_size; ++i) {
375 if (data[i][0] < -threshold)
376 negative++;
377
378 data[i][0] =
379 std::log1p(std::exp((data[i][0] - shift) * multiplier)) / multiplier;
380 }
381 } else {
382 for (Index i = 0; i < data_size; ++i) {
383 if (data[i][0] * multiplier < -threshold)
384 negative++;

386 data[i][0] = std::max(data[i][0] - shift, 0.);
387 }

568 mutable unsigned int negative = std::numeric_limits<unsigned int>::max();

572 ConeOptimizationProblem(
573 const Dune::ParameterTree& config_,
574 const Point& start_,
575 const std::vector<bool>& constrained_,
576 Real shift_,
577 Real threshold_,

579 unsigned int localExtendedDomainSize_,

596
597 Real value(const Point& x, bool subsequent = false) const
598 {
599
600 if (subsequent)
601 return affineVal + coneVal;
602
603 if (!config.template get<bool>("stochastic.stayOnAffine", false)) {
604 current = x;
605 current -= start;
606
607 current.removeUnconstrained(constrained);
608
609 affineVal = 0.5 * (current * current);
610 } else
611 affineVal = 0.;
612
613 current = x;
614 current.forwardTransform();
615 forward++;
616
617 if (config.template get<bool>("stochastic.logSumExp", false)) {
618 negative = 0;
619 Real max = std::numeric_limits<Real>::min();
620 for (unsigned int i = 0; i < localExtendedDomainSize; i++) {
621 if (current.raw()[i][0] < -threshold)
622 negative++;
623 max = std::max(max, -current.raw()[i][0]);
624 }
625 if (logSumExpFactor == 0.)
626 logSumExpFactor = 1. / max;
627 Real sum = 0.;
628 for (unsigned int i = 0; i < localExtendedDomainSize; i++)
629 sum += std::exp(-logSumExpFactor * (current.raw()[i][0] - max));
630
631 coneVal = max + std::log(sum) - std::log(extendedDomainSize);
632 } else {
633 current *= -1.;
634 negative = current.makePositive(shift, threshold, -1.);
635
636 coneVal = 0.5 * (current * current) * extendedDomainSize;
637 }
638
639 return affineVal + coneVal;
640 }

642 Real dir_deriv(const Point& x,
643 const Point& direction,
644 Point& grad,
645 bool subsequent = false) const
646 {
647 if (!config.template get<bool>("stochastic.stayOnAffine", false)) {
648 if (!subsequent) {
649 grad = x;
650 grad -= start;
651
652 grad.removeUnconstrained(constrained);
653 }
654
655 affineVal = 0.5 * (grad * grad);
656 } else {
657 affineVal = 0.;
658
659 grad *= 0.;
660 }
661
662 current = x;
663 current.forwardTransform();
664 forward++;
665
666 if (config.template get<bool>("stochastic.logSumExp", false)) {
667 negative = 0;
668 Real max = std::numeric_limits<Real>::min();
669 for (unsigned int i = 0; i < localExtendedDomainSize; i++) {
670 if (current.raw()[i][0] < -threshold)
671 negative++;
672 max = std::max(max, -current.raw()[i][0]);
673 }
674 if (logSumExpFactor == 0.)
675 logSumExpFactor = 1. / max;
676 Real sum = 0.;
677 for (unsigned int i = 0; i < localExtendedDomainSize; i++) {
678 sum += std::exp(-logSumExpFactor * (current.raw()[i][0] - max));
679 current.raw()[i][0] =
680 logSumExpFactor * std::exp(-logSumExpFactor * current.raw()[i][0]);
681 current.raw()[i][1] = 0.;
682 if (!std::isfinite(current.raw()[i][0]))
683 current.raw()[i][0] = 0.;
684 }
685 current *= 1. / std::exp(max + std::log(sum));
686
687 current.backwardTransform();
688 backward++;
689
690 coneVal = max + std::log(sum) - std::log(extendedDomainSize);
691 } else {
692 current *= -1.;
693 negative = current.makePositive(shift, threshold, -1.);
694
695 current.backwardTransform();
696 backward++;
697
698 coneVal = 0.5 * (current * current);
699 }
700
701 grad -= current;
702
703 if (config.template get<bool>("stochastic.stayOnAffine", false))
704 grad.removeConstrained(constrained);
705
706 return grad * direction;
707 }
