670  void bulkEvaluate(std::vector<typename Traits::RangeType>& output,
671  typename Traits::Indices& sizes) const
672  {
673  // Correctly resize the output container and calculate sizes + strides
674  std::size_t size = 1;
675  typename Traits::Indices strides;
676  std::fill(strides.begin(), strides.end(), 1);
677  for (std::size_t i = 0; i < Traits::dim; ++i) {
678  size *= traits->localCells[i];
679  sizes[i] = traits->localCells[i];
680  for (std::size_t j = i + 1; j < Traits::dim; ++j)
681  strides[j] *= sizes[i];
682  }
683  output.resize(size);
684 
685  // Do the evaluation. The detour to calculating coordinates is unnecessary,
686  // but allows us a minimally invasive approach.
687  for (std::size_t i = 0; i < size; ++i) {
688  // Calculate the location applying above strides
689  std::size_t index = i;
690  typename Traits::DomainType location;
691  for (int j = Traits::dim - 1; j >= 0; --j) {
692  double meshsize = traits->extensions[j] / static_cast<double>(sizes[j]);
693  location[j] =
694  (static_cast<double>(index / strides[j]) + 0.5) * meshsize;
695  index = index % strides[j];
696  }
697 
698  // Perform the evaluation
699  evaluate(location, output[i]);
700  }
