213 void update()
214 {
215 // ensures that FFTW can divide data equally between processes
216 if (cells[dim - 1] % commSize != 0)
217 throw std::runtime_error{
218 "number of cells in last dimension has to be multiple of numProc"
219 };
220 if (dim == 1 && cells[0] % (commSize * commSize) != 0)
221 throw std::runtime_error{
222 "in 1D, number of cells has to be multiple of numProc^2"
223 };
224
225 transposed = config.template get<bool>("fftw.transposed", dim > 1);
226 if (transposed) {
227 // transposed format requires more than one dimension
228 if (dim == 1)
229 transposed = false;
230 // ensures that FFTW can store data in transposed format
231 else if (cells[dim - 2] % commSize != 0) {
232 transposed = false;
233 if (verbose && rank == 0)
234 std::cout
235 << "for transposed transforms second to last dimension has to be"
236 << " multiple of numProc, defaulting to non-transposed"
237 << std::endl;
238 }
239 // avoid R2C transposed format, since it both cuts and transposes first
240 // dim
241 else if (dim == 2) {
242 const std::string& anisotropy =
243 config.template get<std::string>("stochastic.anisotropy", "none");
244
245 // check if R2C will be used
246 if ((std::is_same<typename IsoMatrix<ThisType>::MatrixBackendType,
247 R2CMatrixBackend<ThisType>>::value &&
248 (anisotropy == "none" || anisotropy == "axiparallel")) ||
249 (std::is_same<typename AnisoMatrix<ThisType>::MatrixBackendType,
250 R2CMatrixBackend<ThisType>>::value &&
251 (anisotropy != "none" && anisotropy != "axiparallel")) ||
252 std::is_same<typename IsoMatrix<ThisType>::FieldBackendType,
253 R2CFieldBackend<ThisType>>::value) {
254 transposed = false;
255 if (verbose && rank == 0)
256 std::cout
257 << "R2C backends need more than two dimensions for transposed "
258 "output"
259 << " to avoid confusing layout, defaulting to non-transposed"
260 << std::endl;
261 }
262 }
263 }
264
265 for (unsigned int i = 0; i < dim; i++) {
266 meshsize[i] = extensions[i] / cells[i];
267 extendedCells[i] = embeddingFactor * cells[i];
268 }
269
270 getFFTData(allocLocal, localN0, local0Start);
271
272 for (unsigned int i = 0; i < dim - 1; i++) {
273 localExtendedCells[i] = extendedCells[i];
274 localExtendedOffset[i] = 0;
275 localCells[i] = cells[i];
276 localOffset[i] = 0;
277 }
278 localExtendedCells[dim - 1] = localN0;
279 localExtendedOffset[dim - 1] = local0Start;
280 localCells[dim - 1] = localN0 / embeddingFactor;
281 localOffset[dim - 1] = local0Start / embeddingFactor;
282
283 domainSize = 1;
284 extendedDomainSize = 1;
285 localDomainSize = 1;
286 localExtendedDomainSize = 1;
287 cellVolume = 1.;
288 for (unsigned int i = 0; i < dim; i++) {
289 domainSize *= cells[i];
290 extendedDomainSize *= extendedCells[i];
291 localDomainSize *= localCells[i];
292 localExtendedDomainSize *= localExtendedCells[i];
293 cellVolume *= meshsize[i];
294 }
295
296 if (verbose && rank == 0) {
297 std::cout << "RandomField size: " << localDomainSize << std::endl;
298 std::cout << "RandomField cells: ";
299 for (unsigned int i = 0; i < dim; i++) {
300 std::cout << cells[i] << " ";
301 }
302 std::cout << std::endl;
303 std::cout << "RandomField local cells: ";
304 for (unsigned int i = 0; i < dim; i++) {
305 std::cout << localCells[i] << " ";
306 }
307 std::cout << std::endl;
308 std::cout << "RandomField cell volume: " << cellVolume << std::endl;
309 }
310 }

482class DefaultLoadBalance
483{
484public:
496 void loadbalance(const std::array<int, dim>& size,
497 int P,
498 std::array<int, dim>& dims) const
499 {
500 double opt = 1e100;
501 std::array<int, dim> trydims;
502
503 optimize_dims(dim - 1, size, P, dims, trydims, opt);
504 }
505
506private:
522 void optimize_dims(int i,
523 const std::array<int, dim>& size,
524 int P,
525 std::array<int, dim>& dims,
526 std::array<int, dim>& trydims,
527 double& opt) const
528 {
529 if (i > 0) // test all subdivisions recursively
530 {
531 for (unsigned int k = 1; k <= (unsigned int)P; k++)
532 // dune-randomfield needs exact divisibility
533 if (P % k == 0 && size[i] % k == 0) {
534 // P divisible by k
535 trydims[i] = k;
536 optimize_dims(i - 1, size, P / k, dims, trydims, opt);
537 }
538 } else {
539 // dune-randomfield needs exact divisibility
540 if (size[0] % P == 0) {
541 // found a possible combination
542 trydims[0] = P;
543
544 // check for optimality
545 double m = -1.;
546
547 for (unsigned int k = 0; k < dim; k++) {
548 double mm = ((double)size[k]) / ((double)trydims[k]);
549 if (fmod((double)size[k], (double)trydims[k]) > 0.0001)
550 mm *= 3;
551 if (mm > m)
552 m = mm;
553 }
554 if (m < opt) {
555 opt = m;
556 dims = trydims;
557 }
558 }
559 }
560 }
