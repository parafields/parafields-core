664  void bulkEvaluate(std::vector<typename Traits::RangeType>& output,
665  typename Traits::Indices& sizes) const
666  {
667  // Correctly resize the output container and calculate sizes + strides
668  std::size_t size = 1;
669  typename Traits::Indices strides;
670  std::fill(strides.begin(), strides.end(), 1);
671  for (std::size_t i = 0; i < Traits::dim; ++i) {
672  size *= stochasticPart.localEvalCells[i];
673  sizes[i] = stochasticPart.localEvalCells[i];
674  for (std::size_t j = i + 1; j < Traits::dim; ++j)
675  strides[j] *= sizes[i];
676  }
677  output.resize(size);
678 
679  // Do the evaluation. The detour to calculating coordinates is unnecessary,
680  // but allows us a minimally invasive approach.
681  for (std::size_t i = 0; i < size; ++i) {
682  // Calculate the location applying above strides
683  std::size_t index = i;
684  typename Traits::DomainType location;
685  for (int j = Traits::dim - 1; j >= 0; --j) {
686  location[j] =
687  (static_cast<double>(index / strides[j]) +
688  static_cast<double>(stochasticPart.localEvalOffset[j]) + 0.5) *
689  traits->meshsize[j];
690  index = index % strides[j];
691  }
692 
693  // Perform the evaluation
694  evaluate(location, output[i]);
695  }
