614  void bulkEvaluate(std::vector<typename Traits::RangeType>& output,
615  typename Traits::Indices& sizes) const
616  {
617  // Correctly resize the output container and calculate sizes + strides
618  std::size_t size = 1;
619  typename Traits::Indices strides;
620  std::fill(strides.begin(), strides.end(), 1);
621  for (std::size_t i = 0; i < Traits::dim; ++i) {
622  size *= traits->localCells[i];
623  sizes[i] = traits->localCells[i];
624  for (std::size_t j = i + 1; j < Traits::dim; ++j)
625  strides[j] *= sizes[i];
626  }
627  output.resize(size);
628 
629  // Do the evaluation. The detour to calculating coordinates is unnecessary,
630  // but allows us a minimally invasive approach.
631  for (std::size_t i = 0; i < size; ++i) {
632  // Calculate the location applying above strides
633  std::size_t index = i;
634  typename Traits::DomainType location;
635  for (int j = Traits::dim - 1; j >= 0; --j) {
636  double meshsize = traits->extensions[j] / static_cast<double>(sizes[j]);
637  location[j] =
638  (static_cast<double>(index / strides[j]) + 0.5) * meshsize;
639  index = index % strides[j];
640  }
641 
642  // Perform the evaluation
643  evaluate(location, output[i]);
644  }
645  }
